Orders links to Customers via customer_id and to Products via product_id

Query 1: Customer Order History — uses INNER JOIN

-- Join all 3 tables to see each customer's full order history SELECT c.name AS customer_name, o.order_id, p.product_name, o.quantity, o.total_amount, o.status FROM Customers c JOIN Orders o ON c.customer_id = o.customer_id JOIN Products p ON o.product_id = p.product_id ORDER BY c.name ASC;

Query 2: Highest Value Order — uses Subquery

-- Subquery inside WHERE finds the maximum order total first, -- then the outer query fetches the matching row SELECT c.name AS customer_name, o.order_id, p.product_name, o.total_amount FROM Orders o JOIN Customers c ON o.customer_id = c.customer_id JOIN Products p ON o.product_id = p.product_id WHERE o.total_amount = ( SELECT MAX(total_amount) FROM Orders -- ← this is the subquery );

Query 3: Most Active Customer — uses GROUP BY + ORDER BY

-- Count how many orders each customer placed, then sort by most orders SELECT c.name AS customer_name, COUNT(o.order_id) AS total_orders, SUM(o.total_amount) AS total_spent FROM Customers c JOIN Orders o ON c.customer_id = o.customer_id GROUP BY c.customer_id, c.name ORDER BY total_orders DESC;
